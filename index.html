<script>

let projects = JSON.parse(localStorage.getItem('tf_pro_data') || '[]');

let currentProjectIdx = null;

let signCount = 0;

let signPhotos = {};


function go(id) {

    document.querySelectorAll('.scr').forEach(s => s.classList.remove('active'));

    document.getElementById(id).classList.add('active');

}


function back() { go('s-home'); }


function toast(m) {

    const t = document.getElementById('toast');

    t.textContent = m;

    t.classList.add('show');

    setTimeout(() => t.classList.remove('show'), 2000);

}


function addSign() {

    signCount++;

    const id = `sign-${signCount}`;

    signPhotos[id] = [];

    const div = document.createElement('div');

    div.className = 'fsec';

    div.style.border = '1px solid #eee';

    div.style.borderRadius = '15px';

    div.style.padding = '15px';

    div.style.marginBottom = '15px';

    div.innerHTML = `

        <div style="display:flex; justify-content:space-between; margin-bottom:10px">

            <span style="font-weight:900; color:var(--pr)">#${signCount}</span>

            <span onclick="this.parentElement.parentElement.remove()" style="color:var(--red); font-weight:800; cursor:pointer">SİL</span>

        </div>

        <input class="if" id="${id}-label" placeholder="Tabela Tanımı (Örn: Giriş Kapısı)">

        <select class="if" id="${id}-type">

            <option>Kutu Harf</option><option>Işıklı Tabela</option><option>Vinil Germe</option><option>Cam Folyo</option><option>Diğer</option>

        </select>

        <div style="display:flex; gap:10px; margin-top:10px">

            <input class="if" id="${id}-w" type="number" placeholder="En (m)">

            <input class="if" id="${id}-h" type="number" placeholder="Boy (m)">

        </div>

        <div style="margin-top:10px">

            <input type="file" accept="image/*" multiple onchange="handlePhotos('${id}', this)" style="font-size:12px">

            <div id="${id}-prev" class="pgrid"></div>

        </div>

    `;

    document.getElementById('signs-container').appendChild(div);

}


function handlePhotos(sid, input) {

    Array.from(input.files).forEach(file => {

        const reader = new FileReader();

        reader.onload = e => {

            signPhotos[sid].push(e.target.result);

            renderPrev(sid);

        };

        reader.readAsDataURL(file);

    });

}


function renderPrev(sid) {

    const cont = document.getElementById(`${sid}-prev`);

    cont.innerHTML = signPhotos[sid].map(src => `<div class="pthumb"><img src="${src}"></div>`).join('');

}


function saveProject() {

    const client = document.getElementById('f-client').value;

    const project = document.getElementById('f-project').value;

    if(!client) return toast("Müşteri adı boş olamaz!");


    const signs = [];

    document.querySelectorAll('#signs-container > div').forEach((div, idx) => {

        const id = `sign-${idx + 1}`;

        signs.push({

            label: document.getElementById(`${id}-label`)?.value || 'İsimsiz',

            type: document.getElementById(`${id}-type`)?.value,

            w: document.getElementById(`${id}-w`)?.value,

            h: document.getElementById(`${id}-h`)?.value,

            photos: signPhotos[id] || []

        });

    });


    projects.push({ client, project, signs, date: new Date().toLocaleString('tr-TR') });

    localStorage.setItem('tf_pro_data', JSON.stringify(projects));

    toast("Proje Başarıyla Kaydedildi!");

    renderHome();

    go('s-home');

}


function renderHome() {

    const list = document.getElementById('proj-list');

    if(projects.length === 0) {

        list.innerHTML = `<div style="text-align:center; padding:50px; color:#94a3b8">Henüz proje eklenmemiş.</div>`;

        return;

    }

    list.innerHTML = projects.map((p, i) => `

        <div class="pcard" onclick="openShare(${i})">

            <div style="font-weight:900; font-size:16px">${p.project || 'İsimsiz Proje'}</div>

            <div style="color:var(--txt2); font-size:13px">${p.client}</div>

            <div style="margin-top:10px; display:flex; justify-content:space-between; align-items:center">

                <span style="font-size:11px; background:var(--pr-bg); color:var(--pr); padding:4px 8px; border-radius:6px; font-weight:800">${p.signs.length} Tabela</span>

                <span style="font-size:10px; color:var(--txt3)">${p.date}</span>

            </div>

        </div>

    `).reverse().join('');

}


function openShare(i) {

    currentProjectIdx = i;

    const p = projects[i];

    document.getElementById('sh-title').textContent = p.project || 'İsimsiz Proje';

    document.getElementById('sh-client').textContent = p.client;

    go('s-share');

}


function buildText() {

    const p = projects[currentProjectIdx];

    let text = `*🏗️ TABERA FIELD PROJE RAPORU*\n\n`;

    text += `*Müşteri:* ${p.client}\n`;

    text += `*Proje:* ${p.project}\n`;

    text += `*Tarih:* ${p.date}\n`;

    text += `\n---------------------------\n`;

    

    p.signs.forEach((s, idx) => {

        text += `\n*${idx+1}. ${s.label}*\n`;

        text += `Type: ${s.type}\n`;

        text += `Ölçü: ${s.w || '?'}m x ${s.h || '?'}m\n`;

        if(s.w && s.h) text += `Alan: ${(s.w * s.h).toFixed(2)} m2\n`;

    });

    

    text += `\n_TaberaField Mobile tarafından oluşturuldu._`;

    return text;

}


function shareWhatsApp() {

    const txt = encodeURIComponent(buildText());

    window.open(`https://wa.me/?text=${txt}`, '_blank');

}


function copyText() {

    navigator.clipboard.writeText(buildText());

    toast("Metin kopyalandı!");

}


function printPDF() {

    const p = projects[currentProjectIdx];

    let html = `

        <div class="pdf-header">

            <div><h1 style="color:var(--pr)">TABERA FIELD</h1><p>Saha Ölçüm & Teknik Rapor</p></div>

            <div style="text-align:right"><strong>Tarih:</strong><br>${p.date}</div>

        </div>

        <div class="pdf-grid">

            <div class="pdf-item"><div class="pdf-label">Müşteri / Firma</div><div class="pdf-val">${p.client}</div></div>

            <div class="pdf-item"><div class="pdf-label">Proje Adı</div><div class="pdf-val">${p.project}</div></div>

        </div>

    `;


    p.signs.forEach((s, idx) => {

        const area = (s.w * s.h).toFixed(2);

        html += `

            <div class="pdf-sign">

                <div class="pdf-s-head"><span>${idx+1}. ${s.label}</span><span>${s.type}</span></div>

                <div class="pdf-s-body">

                    <div class="pdf-measures">

                        <div class="pdf-m-box"><div class="pdf-m-val">${s.w || '-'}</div><div class="pdf-label">GENİŞLİK (M)</div></div>

                        <div class="pdf-m-box"><div class="pdf-m-val">${s.h || '-'}</div><div class="pdf-label">YÜKSEKLİK (M)</div></div>

                        <div class="pdf-m-box"><div class="pdf-m-val" style="color:var(--txt)">${area}</div><div class="pdf-label">ALAN (M2)</div></div>

                    </div>

                    <div class="pgrid">

                        ${s.photos.map(src => `<div class="pthumb"><img src="${src}"></div>`).join('')}

                    </div>

                </div>

            </div>

        `;

    });


    document.getElementById('pdf-content').innerHTML = html;

    const doc = document.getElementById('pdf-doc');

    doc.style.display = 'block';

    setTimeout(() => {

        window.print();

        doc.style.display = 'none';

    }, 500);

}


renderHome();

</script>

</body>

</html>
